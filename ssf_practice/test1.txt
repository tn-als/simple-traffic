{

    printf("        ㅡ       ㅡ\n");
    printf("        ㅡ       ㅡ  <--\n");
    printf("        ㅡ       ㅡ\n");
    printf("----------       ----------\n");

void south()
{
    for (int i = 1; i <= 9; i++)
        printf("         l       l\n");
    printf("         l l l l l         \n");
    printf("----------       ----------\n"); 
    printf("        ㅡ       ㅡ  ∧\n");
    for (int i = 1; i <= 2; i++)
        printf("        ㅡ       ㅡ  ㅣ\n");
    printf("----------       ----------\n");
    printf("         l l l l l         \n");
    for (int i = 1; i <= 2; i++)
        printf("         l       l\n");
    printf("         l  ∧   l\n");
    for (int i = 1; i <= 2; i++)
        printf("         l  ㅣ   l\n");
    for (int i = 1; i <= 4; i++)
        printf("         l       l\n");
}

int main() 
{
